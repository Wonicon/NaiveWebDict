用于存放单词卡信息的表word_card
结构：
{
    username varchar(20),//获得卡片的用户名
    details varchar(200),//单词卡细节（词语+解释）
    sender varchar(20),//发送卡片方
}

配置前提：说明文件1的配置已经完成
配置代码
use dict;
create table word_card
(
    username varchar(20),//获得卡片的用户名
    details varchar(200),//单词卡细节（词语+解释）
    sender varchar(20),//发送卡片方
    primary key(user,details,sender)
);
